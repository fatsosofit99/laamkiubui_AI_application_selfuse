def evaluate_accuracy(model: nn.Module, dataloader: DataLoader, device: torch.device) -> float:
    """
    计算模型在给定数据集上的准确率
    """
    model.eval()  # 1. 将模型设置为评估模式 (关键：关闭 Dropout 等训练专属操作)
    correct = 0   # 记录预测正确的总数
    total = 0     # 记录样本总数

    with torch.no_grad():  # 2. 禁用梯度计算 (关键：节省内存并加快计算速度)
        for inputs, labels in dataloader:
            inputs, labels = inputs.to(device), labels.to(device)
            
            # 模型前向传播得到输出 (Logits)
            outputs = model(inputs)
            
            # 3. 找出每个样本概率最大的类别索引作为预测结果
            # torch.max 返回 (最大值, 最大值所在的索引)，我们只需要索引
            _, predicted = torch.max(outputs.data, 1)
            
            # 累计总样本数和正确预测数
            total += labels.size(0)
            correct += (predicted == labels).sum().item()

    # 计算准确率
    accuracy = correct / total
    print(f'准确率{accuracy}')
